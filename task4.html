For the account with the highest number of transactions, make a time series line plot for the behavior of the account over time, including all debits and credits, the different methods, and the with the different categories.
library(ggplot2)
library(plyr)

#read data
transactions = read.csv('data/transactions.csv', header = TRUE)
#get data for time series
most_id = tail(names(sort(table(transactions$account_id))), 1)
df = transactions[transactions$account_id == most_id, ]
df$date = as.Date(df$date)
p <- ggplot(df, aes(x=date, y=balance)) +
  geom_line(  color = "#FC4E07") + 
  theme(axis.text.x=element_text(angle=60, hjust=1)) +
  labs(title="Time series of balance of the account with most transactions",x="Date", y = "balance" )
